-- Script de criação das tabelas para o sistema de imóveis
-- Execute este script no seu banco MySQL

-- Tabela de usuários (admin/editor)
CREATE TABLE IF NOT EXISTS users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  email VARCHAR(255) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  name VARCHAR(255) NOT NULL,
  role ENUM('admin', 'editor') DEFAULT 'editor',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Tabela de imóveis
CREATE TABLE IF NOT EXISTS imoveis (
  id INT AUTO_INCREMENT PRIMARY KEY,
  titulo VARCHAR(255) NOT NULL,
  slug VARCHAR(255) NOT NULL UNIQUE,
  descricao TEXT,
  tipo ENUM('apartamento', 'casa', 'terreno', 'comercial', 'cobertura') NOT NULL,
  status ENUM('aluguel', 'venda') DEFAULT 'aluguel',
  preco DECIMAL(15, 2),
  area_total DECIMAL(10, 2),
  area_construida DECIMAL(10, 2),
  quartos INT DEFAULT 0,
  suites INT DEFAULT 0,
  banheiros INT DEFAULT 0,
  vagas INT DEFAULT 0,
  endereco VARCHAR(255),
  bairro VARCHAR(100),
  cidade VARCHAR(100),
  estado VARCHAR(2),
  cep VARCHAR(10),
  latitude DECIMAL(10, 8),
  longitude DECIMAL(11, 8),
  destaque BOOLEAN DEFAULT FALSE,
  ativo BOOLEAN DEFAULT TRUE,
  created_by INT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (created_by) REFERENCES users(id) ON DELETE SET NULL
);

-- Tabela de imagens dos imóveis
CREATE TABLE IF NOT EXISTS imovel_imagens (
  id INT AUTO_INCREMENT PRIMARY KEY,
  imovel_id INT NOT NULL,
  url VARCHAR(500) NOT NULL,
  alt VARCHAR(255),
  ordem INT DEFAULT 0,
  principal BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (imovel_id) REFERENCES imoveis(id) ON DELETE CASCADE
);

-- Tabela de características dos imóveis
CREATE TABLE IF NOT EXISTS imovel_caracteristicas (
  id INT AUTO_INCREMENT PRIMARY KEY,
  imovel_id INT NOT NULL,
  caracteristica VARCHAR(100) NOT NULL,
  FOREIGN KEY (imovel_id) REFERENCES imoveis(id) ON DELETE CASCADE
);

-- Índices para performance
CREATE INDEX idx_imoveis_tipo ON imoveis(tipo);
CREATE INDEX idx_imoveis_status ON imoveis(status);
CREATE INDEX idx_imoveis_cidade ON imoveis(cidade);
CREATE INDEX idx_imoveis_ativo ON imoveis(ativo);
CREATE INDEX idx_imoveis_destaque ON imoveis(destaque);
CREATE INDEX idx_imovel_imagens_imovel ON imovel_imagens(imovel_id);
CREATE INDEX idx_imovel_caracteristicas_imovel ON imovel_caracteristicas(imovel_id);
